212  bool insert(const edge<Tlabel>& edge) {
213 
214  // Do we need to resize the graph (offset_ container)?
215  if(contains(edge)) return false;
216  if(directed && edge.first >= size()) resize(edge.first + 1);
217  if(!directed && std::max(edge.first, edge.second) >= size()) resize(std::max(edge.first, edge.second) + 1);
218 
219  // Updated adjacency and offset.
220  insertAdjacency(edge);
221  Kokkos::parallel_for("lower_offset_update", offset_.size(),
222  KOKKOS_LAMBDA(const Tlabel i) {
223  offset_(i) += static_cast<Tlabel>(i > edge.first)
224  + static_cast<Tlabel>((i > edge.second)*(!directed));
225  });
226  return true;
227  }

239  void resize(const Tlabel& size) {
240 
241  // Branch based on expansion or contraction of the graph.
242  if(offset_.size() < (size + 1)) {
243  const auto oldsize = offset_.size();
244  Kokkos::resize(offset_, size + 1);
245  Kokkos::parallel_for(offset_.size(), KOKKOS_LAMBDA(const Tlabel i){
246  offset_(i) = i >= oldsize ? offset_(oldsize - 1) : offset_(i);
247  });
248  }
249  else
250  {
251  Kokkos::resize(offset_, size + 1);
252  Kokkos::resize(adjacency_, offset_(offset_.size() - 1));
253  if(directed) return;
254 
255  // Remove any connections to the removed vertices.
256  View temp_offset(offset_.label(), offset_.size());
257  Kokkos::deep_copy(temp_offset, offset_);
258 
259  Tlabel total_offset = 0;
260  Kokkos::parallel_scan("determine_new_offsets", offset_.size() - 1,
261  KOKKOS_LAMBDA(Tlabel i_node, Tlabel& partial_sum, bool is_final) {
262  for(auto i_offset = temp_offset(i_node); i_offset < temp_offset(i_node + 1); ++i_offset) {
263  if(adjacency_(i_offset) < size) partial_sum += 1;
264  }
265  if(is_final) offset_(i_node) = partial_sum;
266  }, total_offset);
267 
268  View temp_adjacency("",adjacency_.size());
269  Kokkos::deep_copy(temp_adjacency, adjacency_);
270  Kokkos::resize(adjacency_, total_offset);
271 
272  Kokkos::parallel_for("update_adjacency", offset_.size() - 1,
273  KOKKOS_LAMBDA(Tlabel i_node) {
274  Tlabel i_offset = 0;
275  for(auto i_adjacency = temp_offset(i_node); i_adjacency < temp_offset(i_node + 1); ++i_adjacency) {
276  if(temp_adjacency(i_adjacency) < size) {
277  adjacency_(i_offset) = temp_adjacency(i_adjacency);
278  i_offset += 1;

301  [[nodiscard]] bool contains(const edge<Tlabel>& check_edge) const
302  {
303  if(empty()) return false;
304  const auto& ordered_edge = !directed ? order_edge(&check_edge)
305  : edge<const Tlabel&>({check_edge.first, check_edge.second});
306  if((!directed ? ordered_edge.second : ordered_edge.first) >= size()) return false;
307  const auto& adjacency = (*this)(ordered_edge.first);
308  for(auto index = 0; index < adjacency.size(); ++index)
309  {
310  if(adjacency(index) == ordered_edge.second) return true;
311  if(adjacency(index) > ordered_edge.second) return false;
312  }
313  return false;
314  }

364  void parallelInit()
365  {
366  if(offset_.size() == 0) {
367  return;
368  }
369 
370  Kokkos::parallel_for("check_correct_acending", offset_.size() - 1,
371  KOKKOS_CLASS_LAMBDA(const Tlabel i_node) {
372  bool is_sorted = true;
373  const int offset_size = offset_(i_node + 1) - offset_(i_node);
374  if(offset_size < 2) return; // nothing to sort
375  for(auto i_check = 0; i_check < (offset_size + 1); ++i_check)
376  {
377  is_sorted = true;
378  for(auto i_offset = offset_(i_node); i_offset < offset_(i_node + 1); ++i_offset)
379  {
380  if(adjacency_(i_offset) > adjacency_(i_offset + 1))
381  {
382  std::swap(adjacency_(i_offset), adjacency_(i_offset + 1));
383  is_sorted = false;
384  }
385  if(is_sorted) break;
386  }
387  if(is_sorted) break;
388  }
389  if(!is_sorted) throw std::runtime_error("Adjacency list could not be sorted.");
390  });
391  }

406  void check_consitency()
407  {
408  // check offset
409  if(offset_.size() == 1) throw std::runtime_error("Offset container must be 0 or greater than 1.");
410  if(offset_.size() != 0 && offset_(0) != 0)
411  throw std::runtime_error("Offset container's first entry must be 0.");
412  if(offset_(offset_.size() - 1) != adjacency_.size())
413  throw std::runtime_error("Offset container is not consistent with adjacency container.");
414 
415  // check undirected graph adjacency.
416  if(!directed)
417  {
418  Kokkos::parallel_for("check_symmetry", offset_.size() - 1,
419  KOKKOS_LAMBDA(const Tlabel i_node)
420  {
421  for(auto i_offset = offset_(i_node); i_offset < offset_(i_node + 1); ++i_offset)
422  {
423  const auto& other_node = adjacency_(i_offset);
424  if(other_node <= i_node) break; // no need to check (checked on other side).
425  if(other_node > offset_.size() - 1)
426  throw std::runtime_error("Undirected Adjacency list contains invalid node index, greater than size.");
427  bool is_found = false;
428  for(auto i_adj_offset = offset_(other_node); i_adj_offset < offset_(other_node + 1); ++i_adj_offset)
429  {
430  if(adjacency_(i_adj_offset) == i_node)
431  {
432  is_found = true;
433  break;
434  }
435  }
436  if(!is_found) throw std::runtime_error("Undirected Adjacency list is not symmetric (degenerate).");
437  }
438  });
439  }

462  void insertAdjacency(const edge<Tlabel>& edge)
463  {
464  const auto ordered_edge = order_edge(&edge);
465  const bool is_adjacency_empty = adjacency_.size() == 0;
466  Kokkos::pair<Tlabel, Tlabel> index_insert = {0, 0};
467 
468  if(!is_adjacency_empty)
469  for(int i_connection = 0; i_connection < 2; ++i_connection)
470  {
471  const Tlabel& connect_0 = i_connection == 0 ? edge.first : edge.second;
472  const Tlabel& connect_1 = i_connection == 0 ? edge.second : edge.first;
473  Tlabel& insert = i_connection == 0 ? index_insert.first : index_insert.second;
474 
475  // Determine offsets, different if the row is empty.
476  if(offset_(connect_0) == offset_(connect_0 + 1)) insert = offset_(connect_0);
477  else insert = offset_(connect_0 + 1);
478  for(auto i_offset = offset_(connect_0); i_offset < offset_(connect_0 + 1); ++i_offset)
479  {
480  if(adjacency_(i_offset) > connect_1)
481  {
482  insert = i_offset;
483  break;
484  }
485  }
486  if(directed) break;
487  }
488 
489  View temp(adjacency_.label(), adjacency_.size());
490  Kokkos::deep_copy(temp, adjacency_);
491  Kokkos::resize(adjacency_, adjacency_.size() + 1 + offset_shift());
492  Kokkos::parallel_for("adjacency_insert", adjacency_.size(),
493  KOKKOS_CLASS_LAMBDA(const Tlabel i)
494  {
495  if(i < index_insert.first) adjacency_(i) = is_adjacency_empty ? 0 : temp(i);
496  else if(i == index_insert.first) adjacency_(i) = edge.second;
497  else if(index_insert.first < i && i < index_insert.second + 1)
498  adjacency_(i) = is_adjacency_empty ? 0 : temp(i - 1);
499  else if(!directed && i == index_insert.second + 1) adjacency_(i) = edge.first;
500  else adjacency_(i) = is_adjacency_empty ? 0 : temp(i - 1 - offset_shift());
501  });
502 
503  }
