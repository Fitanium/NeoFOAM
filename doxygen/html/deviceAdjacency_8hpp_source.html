216  bool insert(const edge<Tlabel>& edge)
217  {
218 
219  // Do we need to resize the graph (offset_ container)?
220  if (contains(edge)) return false;
221  if (directed && edge.first >= size()) resize(edge.first + 1);
222  if (!directed && std::max(edge.first, edge.second) >= size()) resize(std::max(edge.first, edge.second) + 1);
223 
224  // Updated adjacency and offset.
225  insertAdjacency(edge);
226  Kokkos::parallel_for(
227  "lower_offset_update", offset_.size(), KOKKOS_LAMBDA(const Tlabel i) {
228  offset_(i) += static_cast<Tlabel>(i > edge.first)
229  + static_cast<Tlabel>((i > edge.second) * (!directed));
230  }
231  );
232  return true;
233  }

246  {
247 
248  // Branch based on expansion or contraction of the graph.
249  if (offset_.size() < (size + 1))
250  {
251  const auto oldsize = offset_.size();
252  Kokkos::resize(offset_, size + 1);
253  Kokkos::parallel_for(
254  offset_.size(), KOKKOS_LAMBDA(const Tlabel i) {
255  offset_(i) = i >= oldsize ? offset_(oldsize - 1) : offset_(i);
256  }

258  }
259  else

261  Kokkos::resize(offset_, size + 1);

263  if (directed) return;

317  [[nodiscard]] bool contains(const edge<Tlabel>& check_edge) const
318  {
319  if (empty()) return false;
320  const auto& ordered_edge = !directed ? order_edge(&check_edge)
321  : edge<const Tlabel&>({check_edge.first, check_edge.second});
322  if ((!directed ? ordered_edge.second : ordered_edge.first) >= size()) return false;
323  const auto& adjacency = (*this)(ordered_edge.first);
324  for (auto index = 0; index < adjacency.size(); ++index)
325  {
326  if (adjacency(index) == ordered_edge.second) return true;
327  if (adjacency(index) > ordered_edge.second) return false;
328  }
329  return false;
330  }

380  void parallelInit()
381  {
382  if (offset_.size() == 0)
383  {
384  return;

387  Kokkos::parallel_for(
388  "check_correct_acending", offset_.size() - 1, KOKKOS_CLASS_LAMBDA(const Tlabel i_node) {
389  bool is_sorted = true;
390  const int offset_size = offset_(i_node + 1) - offset_(i_node);
391  if (offset_size < 2) return; // nothing to sort
392  for (auto i_check = 0; i_check < (offset_size + 1); ++i_check)
393  {
394  is_sorted = true;
395  for (auto i_offset = offset_(i_node); i_offset < offset_(i_node + 1); ++i_offset)
396  {
397  if (adjacency_(i_offset) > adjacency_(i_offset + 1))
398  {
399  std::swap(adjacency_(i_offset), adjacency_(i_offset + 1));
400  is_sorted = false;
401  }
402  if (is_sorted) break;
403  }
404  if (is_sorted) break;
405  }
406  if (!is_sorted) throw std::runtime_error("Adjacency list could not be sorted.");
407  }

424  void check_consitency()
425  {
426  // check offset
427  if (offset_.size() == 1) throw std::runtime_error("Offset container must be 0 or greater than 1.");
428  if (offset_.size() != 0 && offset_(0) != 0)
429  throw std::runtime_error("Offset container's first entry must be 0.");
430  if (offset_(offset_.size() - 1) != adjacency_.size())
431  throw std::runtime_error("Offset container is not consistent with adjacency container.");
432 
433  // check undirected graph adjacency.
434  if (!directed)
435  {
436  Kokkos::parallel_for(
437  "check_symmetry", offset_.size() - 1, KOKKOS_LAMBDA(const Tlabel i_node) {
438  for (auto i_offset = offset_(i_node); i_offset < offset_(i_node + 1); ++i_offset)
439  {
440  const auto& other_node = adjacency_(i_offset);
441  if (other_node <= i_node) break; // no need to check (checked on other side).
442  if (other_node > offset_.size() - 1)
443  throw std::runtime_error("Undirected Adjacency list contains invalid node index, greater than size.");
444  bool is_found = false;
445  for (auto i_adj_offset = offset_(other_node); i_adj_offset < offset_(other_node + 1); ++i_adj_offset)
446  {
447  if (adjacency_(i_adj_offset) == i_node)
448  {
449  is_found = true;
450  break;
451  }
452  }
453  if (!is_found) throw std::runtime_error("Undirected Adjacency list is not symmetric (degenerate).");
454  }
455  }

480  void insertAdjacency(const edge<Tlabel>& edge)
481  {
482  const auto ordered_edge = order_edge(&edge);
483  const bool is_adjacency_empty = adjacency_.size() == 0;
484  Kokkos::pair<Tlabel, Tlabel> index_insert = {0, 0};
485 
486  if (!is_adjacency_empty)
487  for (int i_connection = 0; i_connection < 2; ++i_connection)
488  {
489  const Tlabel& connect_0 = i_connection == 0 ? edge.first : edge.second;
490  const Tlabel& connect_1 = i_connection == 0 ? edge.second : edge.first;
491  Tlabel& insert = i_connection == 0 ? index_insert.first : index_insert.second;
492 
493  // Determine offsets, different if the row is empty.
494  if (offset_(connect_0) == offset_(connect_0 + 1)) insert = offset_(connect_0);
495  else
496  insert = offset_(connect_0 + 1);
497  for (auto i_offset = offset_(connect_0); i_offset < offset_(connect_0 + 1); ++i_offset)
498  {
499  if (adjacency_(i_offset) > connect_1)
500  {
501  insert = i_offset;
502  break;
503  }
504  }
505  if (directed) break;
506  }
507 
508  View temp(adjacency_.label(), adjacency_.size());
509  Kokkos::deep_copy(temp, adjacency_);
510  Kokkos::resize(adjacency_, adjacency_.size() + 1 + offset_shift());
511  Kokkos::parallel_for(
512  "adjacency_insert", adjacency_.size(), KOKKOS_CLASS_LAMBDA(const Tlabel i) {
513  if (i < index_insert.first) adjacency_(i) = is_adjacency_empty ? 0 : temp(i);
514  else if (i == index_insert.first)
515  adjacency_(i) = edge.second;
516  else if (index_insert.first < i && i < index_insert.second + 1)
517  adjacency_(i) = is_adjacency_empty ? 0 : temp(i - 1);
518  else if (!directed && i == index_insert.second + 1)
519  adjacency_(i) = edge.first;
520  else
521  adjacency_(i) = is_adjacency_empty ? 0 : temp(i - 1 - offset_shift());
522  }
523  );
524  }
